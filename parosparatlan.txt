#include <stdio.h>

int is_even(int n)
{
 return n % 2 == 0;   
}
int is_odd(int n) 
{
return !is_even(n);
}
int main()
{
    int n;
    printf("Adj meg egy szamot!");
    scanf("%d",&n);
    printf("A %d szam paros? %d\n", n, is_even(n));
    printf("A %d szam paratlan? %d\n", n, is_odd(n));
    return 0;
}
